HTML

基础学习

标签

  • 语言标签

    <html lang="zh-CN">

  • 文本格式化标签

    <h1-h6/>
<p>
段落之间有空行    
</p>
<!--换行-->
<br/>
<div>
    纵向区块
</div>
<span>横向区块</span>

  • 图像标签

    <img src="" alt="" title="" width="" height="" border=""></img>

  • 超链接

    <!--_blank则新窗口打开页面，_self则当前窗口打开页面-->
<a herf="http://www.baidu.com" target="_blank"></a>
<!--内部链接可以直接使用路径html文件-->
<a herf="test.html" target="_blank"></a>
<!--空链接-->
<a herf="#" target="_blank"></a>
<!--下载链接-->
<a herf="test.zip" target="_blank"></a>
<!--网页元素链接-->
<a herf="http://www.baidu.com"><img src="img.jpg"></a>
<!--锚点链接 链接到页面内特定位置 herf内容为#则直接返回顶部-->
<a herf="#FirstPosition" target="_blank"></a>
<h1 id ="FirstPosition"></h1>

  • 特殊字符

    特殊字符 描述 字符代码
      空格符 &nbsp;
    < 小于号 &lt;
    > 大于号 &gt;
    & 与符号 &amp;
    ¥ 人民币 &yen;
    ³ 上标 &sup3;

  • 表格标签

    <!--   表格对齐方式                表示是否有边框 单元边沿与内容空白像素 单元格之间像素     表格宽度  表格高度-->
<table align="left/center/right" border="1/0" cellpadding="20" cellspcing="2" width="" height="">
    <!--必须嵌套table内，定义行-->
    <tr>        
        <!--必须嵌套在tr内，定义单元格-->
        <td></td><td></td><td></td>
    </tr>
    <tr>
            <th>排名</th>
            <th>关键词</th>
            <th>趋势</th>
            <th>进入搜索</th>
            <th>最近七日</th>
            <th>相关链接</th>
        </tr>
        <tr>
            <!--跨行合并单元格数量 需要删除合并得单元格-->
            <td colspan="2">1</td>
            <td rolspan>鬼吹灯</td>
            <td><img src="temp.jpg" alt="没有图"></td>
            <td>456</td>
            <td>123</td>
            <td><a href="#">贴吧</a><a href="#">图片</a><a href="#">百科</a</td>
        </tr>
</table>
<!--无序列表-->
<ul>
    <li></li>
</ul>
<!--有序列表 内部元素只可使用li标签-->
<ol>
    <li></li>    
</ol>
<dl>
    <dt>名词</dt>
    <dd>名词解释1</dd>
    <dd>名词解释2</dd>
</dl>

  • 表单标签

    表单一般包括表单域、表单控件、表单信息

    <form action="demo.php" method="POST" name="temp">
    <!--input type可选项-->
    <!--定义可点击按钮，可使用JS启动脚本-->
    "button"
    <!--定义复选框-->
    "checkbox"
    <!--定义输入字段和"浏览"按键，供文件上传-->
    "file"
    <!--定义隐藏的输入字段-->
    "hidden"
    <!--定义图像形式的提交按钮-->
    "image"
    <!--定义密码字段，字符被掩码-->
    "password"
    <!--定义单选按钮-->
    "radio"
    <!--定义重置全部表单数据-->
    "reset"
    <!--定义提交按钮，将表单数据发送服务端 以restful风格&连接法送服务端-->
    "submit"
    <!--定义文本-->
    "text"
    <!-- 表单属性 name为input元素名称 value为表单元素预制值 checked为预制是否选中 maxlength为输入字符最大长度 -->
    <!-- 标签绑定一个表单元素，点击label内文本则自动选定按钮 -->
    <label for="test">用户名:</label><input type="text" id="test" name="username" value="请输入用户名"> <br>
        密码:<input type="password" name="pwd"><br>
    <!-- 单选按钮 -->
    性别：男<input type="radio"> 女<input type="radio"> ???<input type="radio">
    <!-- 多选按钮 -->
    是否同意注册:<input type="checkbox" name="是否">
    <!-- 下拉表单元素 -->
    <select>
        <option>选项1</option>
        <option>选项2</option>
        <option>选项3</option>
        <!-- 默认选项 -->
        <option selected="selected">选项1</option>
    </select>
    <!-- 文本区域 -->
    <textarea></textarea>
</form>
